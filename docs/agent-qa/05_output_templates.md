## Template 2: Unity C# Tests (02_unity_tests.cs)

```csharp
// Unity C# Test Cases — {Feature Name}
// Generated by QA Agent-Tester
// Sprint: S## | Date: YYYY-MM-DD

using NUnit.Framework;
using UnityEngine.TestTools;
using System.Collections;

[TestFixture]
public class UserLoginTests
{
    [SetUp]
    public void SetUp()
    {
        // Reset state trước mỗi test
        AuthManager.ClearSession();
        NetworkManager.SetOnline(true);
    }

    // TC-AUTH-F-001: Login thành công với Google OAuth
    [Test]
    public void LoginSuccess_ValidToken_EntersMainMenu()
    {
        // Arrange
        var mockToken = TestData.VALID_GOOGLE_TOKEN;

        // Act
        var result = AuthManager.LoginWithGoogle(mockToken);

        // Assert
        Assert.IsTrue(result.Success);
        Assert.IsNotNull(result.SessionId);
        Assert.IsNotNull(result.PlayerData);
        Assert.AreEqual(GameState.MainMenu, GameManager.CurrentState);
    }

    // TC-AUTH-F-002: Login với tài khoản bị ban
    [Test]
    public void LoginBannedAccount_ShowsErrorAndBlocksEntry()
    {
        // Arrange
        var mockToken = TestData.BANNED_ACCOUNT_TOKEN;

        // Act
        var result = AuthManager.LoginWithGoogle(mockToken);

        // Assert
        Assert.IsFalse(result.Success);
        Assert.AreEqual("ACCOUNT_BANNED", result.ErrorCode);
        Assert.AreNotEqual(GameState.MainMenu, GameManager.CurrentState);
        Assert.IsTrue(UIManager.IsErrorPopupVisible);
    }

    // TC-AUTH-E-003: Token refresh tự động sau 1 giờ
    [Test]
    public void SessionExpired_AutoRefreshesToken_NoReloginRequired()
    {
        // Arrange
        AuthManager.LoginWithGoogle(TestData.VALID_GOOGLE_TOKEN);
        TimeManager.MockElapsed(61 * 60); // 61 minutes

        // Act
        var result = AuthManager.MakeAuthenticatedRequest("/api/player/profile");

        // Assert
        Assert.IsTrue(result.Success);
        Assert.IsTrue(AuthManager.TokenWasRefreshed);
    }

    [TearDown]
    public void TearDown()
    {
        AuthManager.ClearSession();
        TimeManager.ResetMock();
    }
}

// Test data constants
public static class TestData
{
    public const string VALID_GOOGLE_TOKEN = "mock_valid_google_token_12345";
    public const string BANNED_ACCOUNT_TOKEN = "mock_banned_account_token_67890";
    public const string EXPIRED_TOKEN = "mock_expired_token_99999";
    public const string NEW_PLAYER_TOKEN = "mock_new_player_token_11111";
}
```

## Template 3: PHP PHPUnit Tests (03_php_tests.php)

```php
<?php

// PHP PHPUnit Test Cases — {Feature Name}
// Generated by QA Agent-Tester
// Sprint: S## | Date: YYYY-MM-DD

namespace Tests\Feature;

use Tests\TestCase;
use Illuminate\Foundation\Testing\RefreshDatabase;

class UserLoginTest extends TestCase
{
    use RefreshDatabase;

    protected function setUp(): void
    {
        parent::setUp();
        // Seed test data
        $this->artisan('db:seed', ['--class' => 'TestPlayerSeeder']);
    }

    // TC-AUTH-F-001: Login thành công với Google OAuth
    /** @test */
    public function post_auth_google_valid_token_returns_session(): void
    {
        // Arrange
        $this->mockGoogleOAuth(['email' => 'valid@gmail.com', 'sub' => '123456']);

        // Act
        $response = $this->postJson('/api/auth/google', [
            'google_token' => 'valid_mock_token',
        ]);

        // Assert
        $response->assertStatus(200);
        $response->assertJsonStructure([
            'session_id',
            'player_data' => ['id', 'name', 'level', 'is_new_player'],
        ]);
        $this->assertNotNull($response->json('session_id'));
    }

    // TC-AUTH-F-002: Login với tài khoản bị ban
    /** @test */
    public function post_auth_google_banned_account_returns_403(): void
    {
        // Arrange
        $this->mockGoogleOAuth(['email' => 'banned@gmail.com', 'sub' => 'banned_123']);

        // Act
        $response = $this->postJson('/api/auth/google', [
            'google_token' => 'banned_mock_token',
        ]);

        // Assert
        $response->assertStatus(403);
        $response->assertJson(['error' => 'ACCOUNT_BANNED']);
    }

    // TC-AUTH-F-003: Refresh token
    /** @test */
    public function post_auth_refresh_expired_session_returns_new_token(): void
    {
        // Arrange
        $session = $this->createExpiredSession();

        // Act
        $response = $this->postJson('/api/auth/refresh', [
            'session_id' => $session->id,
        ]);

        // Assert
        $response->assertStatus(200);
        $response->assertJsonStructure(['new_session_id', 'expires_at']);
        $this->assertNotEquals($session->id, $response->json('new_session_id'));
    }

    // TC-API-S-001: Không có token → 401
    /** @test */
    public function get_player_profile_without_auth_returns_401(): void
    {
        $response = $this->getJson('/api/player/profile');
        $response->assertStatus(401);
    }

    // Helpers
    protected function mockGoogleOAuth(array $data): void
    {
        \Google_Client::shouldReceive('verifyIdToken')->andReturn($data);
    }

    protected function createExpiredSession(): object
    {
        return \DB::table('sessions')->insertGetId([
            'player_id' => 1,
            'token' => 'expired_token_999',
            'expires_at' => now()->subHours(25),
        ]);
    }
}
```
